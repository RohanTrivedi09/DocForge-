### Body text style
```python
from docx.shared import Pt, RGBColor
from docx.oxml import OxmlElement
from docx.oxml.ns import qn

def apply_body_style(paragraph):
    paragraph.alignment = WD_ALIGN_PARAGRAPH.JUSTIFY
    pf = paragraph.paragraph_format
    pf.line_spacing = Pt(12 * 1.5)   # 1.5 × 12pt
    pf.space_after  = Pt(6)
    pf.space_before = Pt(0)
    pf.first_line_indent = Pt(0)
    for run in paragraph.runs:
        run.font.name = 'Times New Roman'
        run.font.size = Pt(12)
        run.font.color.rgb = RGBColor(0x00, 0x00, 0x00)
```

### Heading colors — XML-level enforcement (non-negotiable)
```python
HEADING_SPEC = {
    1: {'size': 14, 'color': '000080', 'bold': True},
    2: {'size': 14, 'color': 'C00000', 'bold': True},
    3: {'size': 12, 'color': '1F4D78', 'bold': True},
}

def apply_heading_sop(paragraph, level):
    spec = HEADING_SPEC.get(level, HEADING_SPEC[3])
    paragraph.alignment = WD_ALIGN_PARAGRAPH.LEFT
    pf = paragraph.paragraph_format
    pf.space_before = Pt(12)
    pf.space_after  = Pt(6)
    for run in paragraph.runs:
        run.font.name  = 'Times New Roman'
        run.font.size  = Pt(spec['size'])
        run.font.bold  = spec['bold']
        # XML-level color — bypasses style inheritance
        run.font.color.rgb = RGBColor.from_string(spec['color'])
        rPr = run._r.get_or_add_rPr()
        for existing in rPr.findall(qn('w:color')):
            rPr.remove(existing)
        color_el = OxmlElement('w:color')
        color_el.set(qn('w:val'), spec['color'].upper())
        rPr.append(color_el)
```

def add_sop_table(doc, headers, rows):
    table = doc.add_table(rows=1 + len(rows), cols=len(headers))
    table.style = 'Table Grid'

    # Full width = 9026 twips (A4, 1" margins)
    tbl = table._tbl
    tblPr = tbl.find(qn('w:tblPr')) or OxmlElement('w:tblPr')
    tblW = OxmlElement('w:tblW')
    tblW.set(qn('w:w'), '9026')
    tblW.set(qn('w:type'), 'dxa')
    tblPr.append(tblW)

    def set_cell_shading(cell, fill_hex):
        tc_pr = cell._tc.get_or_add_tcPr()
        shd = OxmlElement('w:shd')
        shd.set(qn('w:val'),   'clear')
        shd.set(qn('w:color'), 'auto')
        shd.set(qn('w:fill'),  fill_hex.lstrip('#').upper())
        tc_pr.append(shd)

    def set_cell_padding(cell):
        tc_pr = cell._tc.get_or_add_tcPr()
        mar = OxmlElement('w:tcMar')
        for side in ['top', 'bottom']:
            el = OxmlElement(f'w:{side}')
            el.set(qn('w:w'), '80')
            el.set(qn('w:type'), 'dxa')
            mar.append(el)
        for side in ['left', 'right']:
            el = OxmlElement(f'w:{side}')
            el.set(qn('w:w'), '120')
            el.set(qn('w:type'), 'dxa')
            mar.append(el)
        tc_pr.append(mar)

    # Header row — #2C5282 fill, white bold text
    hdr_row = table.rows[0]
    for i, h in enumerate(headers):
        cell = hdr_row.cells[i]
        cell.text = ''
        set_cell_shading(cell, '2C5282')
        set_cell_padding(cell)
        run = cell.paragraphs[0].add_run(h)
        run.font.name  = 'Times New Roman'
        run.font.size  = Pt(10)
        run.font.bold  = True
        run.font.color.rgb = RGBColor(0xFF, 0xFF, 0xFF)

    # Body rows — #F7FAFC fill, black text
    for r_idx, row_data in enumerate(rows):
        row = table.rows[r_idx + 1]
        for c_idx, val in enumerate(row_data):
            cell = row.cells[c_idx]
            cell.text = ''
            set_cell_shading(cell, 'F7FAFC')
            set_cell_padding(cell)
            run = cell.paragraphs[0].add_run(str(val))
            run.font.name  = 'Times New Roman'
            run.font.size  = Pt(10)
            run.font.color.rgb = RGBColor(0x00, 0x00, 0x00)
```

import zipfile, re

def extract_and_verify_colors(docx_path):
    """Unpack docx XML and extract all hex color values used."""
    found_colors = {}
    with zipfile.ZipFile(docx_path) as z:
        for name in z.namelist():
            if name.endswith('.xml'):
                content = z.read(name).decode('utf-8', errors='ignore')
                colors = re.findall(r'w:color[^/]*/?>.*?w:val="([0-9A-Fa-f]{6})"', content)
                for c in colors:
                    found_colors[c.upper()] = found_colors.get(c.upper(), 0) + 1
    return found_colors
    # Log these — compare against spec before generating
    # If mismatch found, return warning to frontend before proceeding
```

## 6. Validation after generation

After generating the docx, run a post-generation check:
```python
def validate_sop_output(doc):
    issues = []
    para_count = len(doc.paragraphs)
    h1_count = sum(1 for p in doc.paragraphs if p.style.name == 'Heading 1')
    h2_count = sum(1 for p in doc.paragraphs if p.style.name == 'Heading 2')
    # Check heading colors on first run of each heading
    for p in doc.paragraphs:
        if p.style.name in ('Heading 1', 'Heading 2', 'Heading 3'):
            if p.runs:
                color = p.runs[0].font.color.rgb
                expected = {
                    'Heading 1': '000080',
                    'Heading 2': 'C00000',
                    'Heading 3': '1F4D78'
                }[p.style.name]
                if str(color).upper() != expected.upper():
                    issues.append(f'{p.style.name} color mismatch: got {color}, expected {expected}')
    return {
        'paragraph_count': para_count,
        'h1_count': h1_count,
        'h2_count': h2_count,
        'issues': issues
    }
```
